minikube delete

minikube start --docker-env HTTP_PROXY=${http_proxy} --docker-env HTTPS_PROXY=${https_proxy} --docker-env NO_PROXY=192.168.99.0/24

minikube ssh -- systemctl show docker --property=Environment --no-pager
